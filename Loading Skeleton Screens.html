Creative Design
Experience the power of modern web design with stunning visuals and smooth animations.
Our skeleton screens provide a better loading experience, keeping users engaged while content loads.
Transform your user interface with these beautiful loading states. 
// Skeleton Loading Animation
.skeleton {
    background: linear-gradient(90deg, 
        rgba(255, 255, 255, 0.1) 25%, 
        rgba(255, 255, 255, 0.2) 50%, 
        rgba(255, 255, 255, 0.1) 75%);
    background-size: 200% 100%;
    animation: loading 1.5s infinite;
}

@keyframes loading {
    0% {
        background-position: 200% 0;
    }
    100% {
        background-position: -200% 0;
    }
}

// Fade in content when loaded
function showContent() {
    document.querySelectorAll('.content').forEach((content, index) => {
        setTimeout(() => {
            content.classList.add('show');
        }, index * 200);
    });
}

// Hide skeleton and show content
function loadContent() {
    document.querySelectorAll('.skeleton-elements').forEach(element => {
        element.style.display = 'none';
    });
    showContent();
}
