// Particle Text Reveal
class ParticleText {
    constructor(canvas, text) {
        this.canvas = canvas;
        this.ctx = canvas.getContext('2d');
        this.text = text;
        this.particles = [];
        this.mouse = { x: null, y: null };
        this.effect = 'explode';
        this.speed = 1;
        
        this.resize();
        this.createParticles();
        this.animate();
        
        window.addEventListener('resize', () => this.resize());
        canvas.addEventListener('mousemove', (e) => {
            this.mouse.x = e.x;
            this.mouse.y = e.y;
        });
    }
    
    createParticles() {
        // Create particles from text
        const tempCanvas = document.createElement('canvas');
        const tempCtx = tempCanvas.getContext('2d');
        
        tempCtx.font = 'bold 80px Arial Black';
        const textWidth = tempCtx.measureText(this.text).width;
        
        tempCanvas.width = textWidth;
        tempCanvas.height = 100;
        
        tempCtx.font = 'bold 80px Arial Black';
        tempCtx.fillStyle = 'white';
        tempCtx.fillText(this.text, 0, 80);
        
        const imageData = tempCtx.getImageData(0, 0, textWidth, 100);
        const data = imageData.data;
        
        for (let y = 0; y < 100; y += 2) {
            for (let x = 0; x < textWidth; x += 2) {
                const index = (y * textWidth + x) * 4;
                const alpha = data[index + 3];
                
                if (alpha > 128) {
                    this.particles.push(new Particle(
                        x + (this.canvas.width - textWidth) / 2,
                        y + (this.canvas.height - 100) / 2
                    ));
                }
            }
        }
    }
}
