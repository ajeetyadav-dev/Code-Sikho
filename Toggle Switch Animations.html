// Toggle Switch Animations
// Add event listeners to all toggles
document.querySelectorAll('input[type="checkbox"]').forEach(toggle => {
    toggle.addEventListener('change', function() {
        const status = document.getElementById(this.id + 'Status');
        
        if (this.checked) {
            status.textContent = 'ON';
            status.classList.add('active');
            
            // Add special effects for specific toggles
            if (this.id === 'icon') {
                status.textContent = 'Night';
            }
        } else {
            status.textContent = 'OFF';
            status.classList.remove('active');
            
            if (this.id === 'icon') {
                status.textContent = 'Day';
            }
        }
    });
});

// Toggle all switches with animation
function toggleAll() {
    const toggles = document.querySelectorAll('input[type="checkbox"]');
    toggles.forEach((toggle, index) => {
        setTimeout(() => {
            toggle.checked = !toggle.checked;
            toggle.dispatchEvent(new Event('change'));
        }, index * 100);
    });
}
